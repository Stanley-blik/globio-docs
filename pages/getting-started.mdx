### Step 5: Initialize Globio

```javascript
// Web/React/Node.js
import { GlobalDoc } from '@stanlink/globio'

const globio = new GlobalDoc({
  apiKey: 'your-api-key-here',
  projectId: 'your-project-id'
})

// Start using the database
const users = globio.collection('users')
await users.add({
  name: 'John Doe',
  level: 1,
  score: 0
})
```
